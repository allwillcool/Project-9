unsigned int count_steps(unsigned int x, unsigned int y) { 
    unsigned int count = 0; // Ћ≥чильник крок≥в, ≥н≥ц≥ал≥зований нулем 
     
    while (x != y) { // ѕоки x не дор≥внюЇ y, виконуЇмо цикл 
        if (x < y) { // якщо x менше y 
            if ((y - x) == 1 || (y - x) % 2 == 0) { 
                // якщо р≥зниц€ м≥ж x та y Ї 1 або парним числом, зб≥льшуЇмо x на 1 
                x++; 
            } else { 
                // якщо р≥зниц€ м≥ж x та y Ї непарним числом, зб≥льшуЇмо x на 2 
                x += 2; 
            } 
        } else { // якщо x б≥льше y 
            if ((x - y) == 1 || (x - y) % 2 == 0) { 
                // якщо р≥зниц€ м≥ж x та y Ї 1 або парним числом, зменшуЇмо x на 1 
                x--; 
            } else { 
                // якщо р≥зниц€ м≥ж x та y Ї непарним числом, зменшуЇмо x на 2 
                x -= 2; 
            } 
        } 
         
        count++; // «б≥льшуЇмо л≥чильник крок≥в на 1 
    } 
     
    return count; // ѕовертаЇмо к≥льк≥сть крок≥в 
}
